[docs]    def ix(self, word):
        """
        Returns the index on self.vocab and self.clusters for `word`
        """
        temp = np.where(self.vocab == word)[0]
        if temp.size == 0:
            raise KeyError('Word not in vocabulary')
        else:
            return temp[0]

        """

        """
        idx = self.ix(word)
        return self.clusters[idx]
